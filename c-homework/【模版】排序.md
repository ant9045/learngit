#include <bits/stdc++.h>
using namespace std;

int a[100000], tmp[100000];

void merge(int left, int mid, int right);
void mergechaifen(int left, int right);

int main() {
    int n;
    cin >> n;
    int nonzero_count = 0;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        if (a[i]!= 0) {
            a[nonzero_count++] = a[i];
        }
    }

    mergechaifen(0, nonzero_count - 1);

    for (int i = 0; i < nonzero_count; i++)
        cout << a[i] << " ";

    return 0;
}

void merge(int left, int mid, int right) {
    int i = left, j = mid + 1, k = left;
    while (i <= mid && j <= right) {
        if (a[i] <= a[j])
            tmp[k++] = a[i++];
        else
            tmp[k++] = a[j++];
    }
    while (i <= mid)
        tmp[k++] = a[i++];
    while (j <= right)
        tmp[k++] = a[j++];
    for (int i = left; i <= right; i++)
        a[i] = tmp[i];
}

void mergechaifen(int left, int right) {
    if (left >= right)
        return;
    int mid = (left + right) / 2;
    mergechaifen(left, mid);
    mergechaifen(mid + 1, right);
    merge(left, mid, right);
}
