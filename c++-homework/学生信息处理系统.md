学生信息处理系统
```c
#include <iostream>
#include <string>
#include <cstdio>
#include <cstring>
#include <sstream>
#include <cstdlib>
using namespace std;

class Student {
private:
    string name;
    int age;
    int student_id;
    int grades[4];
    float average_grade;

public:
    void input() {
        string input_str;
        getline(cin, input_str);
        stringstream ss(input_str);
        getline(ss, name, ',');
        ss >> age;
        ss.ignore();
        ss >> student_id;
        ss.ignore();
        for (int i = 0; i < 4; ++i) {
            ss >> grades[i];
            ss.ignore();
        }
    }

    void calculate() {
        int sum = 0;
        for (int i = 0; i < 4; ++i) {
            sum += grades[i];
        }
        average_grade = static_cast<float>(sum) / 4;
    }

    void output() {
        cout << name << "," << age << "," << student_id << "," << average_grade << endl;
    }
};

int main() {
    Student student;
    student.input();
    student.calculate();
    student.output();
    return 0;
}
```
